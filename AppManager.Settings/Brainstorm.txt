// Example usage in your application
var actionManager = new ActionManager();

// Launch an application
var launchParams = new ActionParameters 
{ 
    ExecutablePath = @"C:\Program Files\Example\example.exe",
    Arguments = "--start-minimized"
};
bool launched = await actionManager.ExecuteActionAsync("Launch", "example", launchParams);

// Close an application with force
var closeParams = new ActionParameters 
{ 
    ForceOperation = true,
    IncludeChildProcesses = true 
};
bool closed = await actionManager.ExecuteActionAsync("Close", "example", closeParams);

// Focus a specific window
var focusParams = new ActionParameters 
{ 
    WindowTitle = "Document1 - Notepad" 
};
bool focused = await actionManager.ExecuteActionAsync("Focus", "notepad", focusParams);


// Enum for action types
var actionManager = new ActionManager();

// Using enum (preferred)
await actionManager.ExecuteActionAsync(AppActionEnum.Launch, "notepad");

// Using string (backward compatibility)
await actionManager.ExecuteActionAsync("Launch", "notepad");
